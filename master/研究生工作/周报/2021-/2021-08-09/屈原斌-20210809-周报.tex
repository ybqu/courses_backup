\section{本周部分重点工作详述}

\subsection{特征实验更新}
\begin{itemize}
  \item 测试集：
  \begin{itemize}
    \item $Dataset_1$: 395篇，不离题:部分离题:离题=197:125:73
    \item $Dataset_1$: 1000篇，不离题:部分离题:离题=500:300:200
  \end{itemize}
  \item 方案：
  \begin{itemize}
    \item 抽取特征进行三分类
    \item 特征：
    \begin{itemize}
      \item [1.] 标题-正文相似度：TFIDF、Doc2vec、Word2vec、IDF-Embedding、BERTGen(生成标题与原标题)
      \item [2.] 标题关键词
      \begin{itemize}
        \item 关键词在正文中出现的比例
        \item 关键词排名位置（是否出现在top5）
        \item 关键词是否出现在开头/结尾
        \item 包含所有关键词句子比例
      \end{itemize}
      \item [3.] 正文关键词（TFIDF值排序Top5）
      \begin{itemize}
        \item 关键词与标题相似度
      \end{itemize}
      \item [4.] BertNSP特征向量
    \end{itemize}
  \end{itemize}
  \item 指标更新，见表1、表2
  \item 结论：
  \begin{itemize}
    \item 使用特征Marco-F1指标最优
    \item 特征方案SVC分类器结果最优
    \item 特征+BertNSP特征对$Dataset_2$指标影响较大
  \end{itemize}
\end{itemize}
